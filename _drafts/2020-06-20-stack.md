# 用python实现栈

```python
class Stack:
    def __init__(self):
        self.items = []
    
    def is_empty(self):
        return bool(self.items)
    
    def push(self, item):
        self.items.append(item)
        
    def peek(self):
        self.items[-1]
        
    def size(self):
        return len(self.items)
    
    def pop(self):
        return self.items.pop()
```
